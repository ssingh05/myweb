var tOneForCount = function(rows, columns, total) {
    this.rows = rows;
    this.columns = columns;
    this.numeber = total;
    this.getCelloCount = function() {
        return  this.columns + this.rows + this.numeber;
    }
}

// type two

function tTwoForCount(rows, columns){
    this.rows = rows;
    this.column = columns;
    this.getCount = function(){
        return this.rows *this.column; 
    }
}

// for function one
var tables = new tOneForCount(10, 152, 10);
var cellnumber = tables.getCelloCount();

// for functiuon two

var t = new tTwoForCount(5,10);
var count = t.getCount();
